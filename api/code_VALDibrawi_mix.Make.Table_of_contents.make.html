let make document_for_toc =
      Params.map_output
        ~html:(fun ()->
          sprintf  "{bypass endbypass}\n<hr/>
                    <b>Table of Contents:</b>\n%s\n<hr/>\n{endbypass}"
            (Dibrawi.Brtx_transform.html_toc 
               (Dibrawi.Preprocessor.brtx2brtx document_for_toc)))
        ~latex:(fun () ->
          "{bypass}\n\\vspace{1em}\\hrule\n\\tableofcontents\n\\vspace{1em}\\hrule\n\n{end}")
