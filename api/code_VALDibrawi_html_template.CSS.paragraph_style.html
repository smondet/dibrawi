let paragraph_style ?(debug=false) ?(separate="0.5em") : component =
    fun params ->
    cat ~sep:new_line [
      str "div.p {";
      str "    padding-bottom: 0em;";
      if debug then 
        str "/* The debug border: */ border: thin #0e0 solid;"
      else 
        empty;
      str "}";
      str $ sprintf "div.p + div.p { padding-top: %s; }" separate;
    ]
